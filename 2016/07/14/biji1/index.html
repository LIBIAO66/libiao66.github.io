  1. onKeyup=”textKey()”触发事件

    控制输入字数 最多只能输入1000个字

  var busiPlanOutline = $("#demandContent").val();

}

5 表单验证之validate

$(document).ready(function() {
        //聚焦第一个输入框

        $("#enterpriseName").focus();
        //#mainForm为需要进行验证的表单ID
        $("#mainForm").validate({
             //验证规则
            rules : {
                enterpriseName : {  // 需要进行验证的输入框name
                    required : true  // 验证条件：必填
                },
                enterpriseType : {
                    required : true
                },
                linkman : {
                    required : true
                },
                companyPosition : {
                    required : true
                },
                linkeway : {
                    required : true,
                    isTel : true // 验证条件：联系方式验证    
                },
                demandContent : {
                    required : true
                },
                demandBudget : {
                    required : true
                },
                startTime : {
                    required : true
                },
                endTime : {
                    required : true
                }
            },
            messages : {
                enterpriseName : {
                    required : "请输入企业名称"  // 验证未通过的消息
                },
                enterpriseType : {
                    required : "请选择企业组织形式"
                },
                linkman : {
                    required : "请输入联系人"
                },
                companyPosition : {
                    required : "请输入公司职位"
                },
                linkeway : {
                    required : "请输入身份证",
                    isZipCode : "请正确填写您的身份证"
                },
                demandContent : {
                    required : "必填"
                },
                demandBudget : {
                    required : "请输入预算范围"
                },
                startTime : {
                    required : "请选择发布开始时间"
                },
                endTime : {
                    required : "请选择发布结束时间"
                }
            },
              //显示验证错误
            errorPlacement: function(error, element) {
             if ( element.is(":radio") )
            error.appendTo( element.parent().parent());

            else if(element.is(":text"))
            error.appendTo( element.parent().parent() );
            else
            error.appendTo( element.parent());
           }
        });
    });
